VBoxManage createvm --name ubuntu --ostype "Ubuntu_64" --register --basefolder .\VM\

VBoxManage modifyvm ubuntu --ioapic on

VBoxManage modifyvm ubuntu --cpus 1

VBoxManage modifyvm ubuntu --cpus 2

VBoxManage modifyvm ubuntu --memory 1536

VBoxManage modifyvm ubuntu --vram 128

VBoxManage modifyvm ubuntu --nic1 nat

VBoxManage createhd --filename ./VM/ubuntu/ubuntu_DISK.vdi --size 20000 --format VDI

VBoxManage storagectl ubuntu --name "SATA Controller" --add sata --controller IntelAhci

VBoxManage storageattach ubuntu --storagectl "SATA Controller" --port 0 --device 0 --type hdd --medium  ./VM/ubuntu/ubuntu_DISK.vdi

VBoxManage storagectl ubunt --name "IDE Controller" --add ide --controller PIIX4

VBoxManage storagectl ubuntu --name "IDE Controller" --add ide --controller PIIX4

VBoxManage storageattach ubuntu --storagectl "IDE Controller" --port 1 --device 0 --type dvddrive --medium ./VM/debian.iso

VBoxManage storageattach ubuntu --storagectl "IDE Controller" --port 1 --device 0 --type dvddrive --medium "C:\Users\LENOVO\.VirtualBox\VM\debian.iso"

VBoxManage modifyvm ubuntu --boot1 dvd --boot2 disk --boot3 none --boot4 none

VBoxManage startvm ubuntu
